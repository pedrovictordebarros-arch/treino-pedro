  <script>
    const exercises = ["Agachamento", "Flexão", "Prancha", "Afundo", "Mountain Climbers"];
    let current = 0;
    let isRest = false;
    let timeLeft = 40;
    let timer;

    function updateDisplay() {
      document.getElementById("timer").textContent =
        (timeLeft < 10 ? "00:0" : "00:") + timeLeft;
    }

    function startWorkout() {
      current = 0;
      isRest = false;
      timeLeft = 40;
      clearInterval(timer);
      timer = setInterval(runTimer, 1000);
    }

    function runTimer() {
      updateDisplay();
      timeLeft--;

      if (timeLeft < 0) {
        if (!isRest) {
          // troca para descanso
          isRest = true;
          timeLeft = 20;
          document.getElementById("timer").textContent = "Descanso";
        } else {
          // próximo exercício
          isRest = false;
          current++;
          if (current >= exercises.length) {
            clearInterval(timer);
            document.getElementById("timer").textContent = "Treino Finalizado!";
            return;
          }
          timeLeft = 40;
          alert("Próximo: " + exercises[current]);
        }
      }
    }
  </script>
